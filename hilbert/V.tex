\chapter{Foundations of the Verification Tensor}

\begin{axiom}[Working Basis]
\label{ax:foundation}
We work within the Zermelo--Fraenkel set theory with the Axiom of Choice (ZFC).
Martin’s Axiom (MA) is assumed for all countable chain condition posets,
and we impose the \emph{Axiom of Event Selection (Boltzmann)} restricting
attention to locally finite, realizable event histories.
\end{axiom}

\begin{definition}[Core Objects and Contracts]
\label{def:coreobjects}
Let the following tensors and maps constitute the primitive objects:

\begin{enumerate}
  \item $\U$: the \textbf{Universe tensor}, a measurable causal order supporting curvature and action evaluation.
  \item $\mathbf{E}_k$: an \textbf{Event tensor}, a finite or $\sigma$-finite subset of $\U$ consistent with Event Selection.
  \item $\Acal$: the \textbf{Informational Curvature Action}, $\Acal:\mathrm{Histories}\to\mathbb{R}\cup\{+\infty\}$,
        lower semicontinuous under MA; its extremals encode reciprocity.
  \item $\Bform$: the \textbf{Reciprocity Bilinear Form}, a symmetric bilinear form on the tangent space
        $\mathcal{V}$ to admissible histories, non-degenerate on the quotient by gauge.
  \item $\InjectiveMap$: the \textbf{Injective Class Map},
        $\InjectiveMap:\mathcal{C}/\!\sim\;\hookrightarrow\;\mathrm{Rep}$,
        injective and order-reflecting on equivalence classes of realizable histories.
\end{enumerate}
Each object satisfies its \emph{contract}: a list of pre- and post-conditions
ensuring compatibility with Axiom~\ref{ax:foundation}.
\end{definition}

\section{Keystone Results}

\begin{theorem}[Reciprocity–Action Equivalence]
\label{thm:reciprocity-action}
Let $\Acal$ be the curvature/action functional on histories admissible under
Event Selection. Under MA, any extremal of $\Acal$ induces a symmetric bilinear
form $\Bform$ on the tangent space of admissible perturbations such that:
\begin{enumerate}[(i)]
  \item $\Bform$ is non-degenerate on the quotient by gauge;
  \item Stationarity of $\Acal$ is equivalent to reciprocity,
        $\Bform(\delta_1,\delta_2)=\Bform(\delta_2,\delta_1)$,
        and to the Euler–Lagrange equations on the quotient space.
\end{enumerate}
\end{theorem}

\begin{lemma}[Injective Class Map]
\label{lem:injective}
The map $\InjectiveMap$ from equivalence classes of realizable histories (modulo gauge)
to representable objects is injective and order-reflecting:
\[
 (x \not\sim y)\Rightarrow
   \InjectiveMap([x])\neq \InjectiveMap([y]),\qquad
 x\preceq y \Rightarrow
   \InjectiveMap([x])\preceq \InjectiveMap([y]).
\]
\end{lemma}

\begin{theorem}[Coherence Implies Entropic Monotonicity]
\label{thm:coherence-entropy}
Assuming Axiom~\ref{ax:foundation} and the coherence constraint
$U^{(4)} = 0$, the induced evolution on admissible histories satisfies
$\Delta S \ge 0$ for the entropy functional $S$ defined on equivalence classes,
with equality if and only if the reciprocity/gauge class is stationary.
\end{theorem}

\section{Verification Workflow}

\begin{definition}[Verification Steps]
\label{def:verify-steps}
Given a target file $\mathsf{F}$ (e.g.\ $P_{\text{clean}}$, $G1S$, or $G2S$),
verification proceeds in three stages:
\begin{enumerate}
  \item \textbf{Type Check:} confirm that each object in Definition~\ref{def:coreobjects}
        appears in $\mathsf{F}$ with its declared contract.
  \item \textbf{Obligation Check:} ensure $\mathsf{F}$ discharges the hypotheses
        of Theorems~\ref{thm:reciprocity-action}--\ref{thm:coherence-entropy}.
  \item \textbf{Coherence Check:} verify that $\mathsf{F}$ preserves the
        entropic monotonicity $\Delta S\ge0$ under its constructions.
\end{enumerate}
\end{definition}
